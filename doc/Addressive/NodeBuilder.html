def app(app, options = {})
  a = app.to_app
  @node.apps << a
  if a.respond_to? :generate_uri_specs
    a.generate_uri_specs(options).each do |k,v|
      specs = @node.uri_spec(k)
      specs << specs.convert(v).each{|spec| spec.app = a }
    end
  end
end
