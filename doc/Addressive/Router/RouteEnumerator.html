def each
  total = @routes.size
  scanned = 0
  scan_time = 0
  @routes.each do |spec|
    scanned += 1
    t = Time.now
    vars = spec.template.extract( spec.template.absolute? ? @url : @path)
    scan_time += (Time.now - t)
    if vars
      node, action = @actions[spec];
      yield Addressive.new(node, action, vars, spec, {:routes.scanned'=>scanned,:routes.total'=>total,:duration => scan_time})
    end
  end

end
