def <<(spec)
  if spec.template.absolute?
    @partitions[0] << spec
    @partitions[1] << spec
  else
    tkns = spec.template.send(:tokens)
    if tkns.empty?
      # emtpy uri, no problem?!
      @partitions[0] << spec
      @partitions[1] << spec
      return
    end
    tk = tkns.first
    if tk.literal?
      if tk.string.start_with?(@prefix)
        @partitions[0] << spec
      elsif @prefix.start_with?(tk.string) and tkns[1]
        @partitions[0] << spec
        @partitions[1] << spec
      else
        @partitions[1] << spec
      end
    else
      @partitions[0] << spec
      @partitions[1] << spec
    end
  end
  
end
