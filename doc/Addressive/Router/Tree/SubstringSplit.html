def <<(spec)
  if spec.template.send(:tokens).select(&:literal?).none?{|tk| tk.string.include? @substring }
    # okay, this template does not require the substring.
    # so it can be matched, even when the substring is missing
    @partitions[1] << spec
  end
  @partitions[0] << spec
end
