def <<(*args)
  args = args.flatten
  fails = args.select{|a| !a.kind_of? URISpec }
  raise ArgumentError, "Expected to receive only URISpecs but, got #{fails.map(&:inspect).join(', ')}" if fails.size != 0
  @specs.push( *args )
end
